### Standard Ads Parameters

<ParamField body="campaignType" type="string" required>
	For Conversions Goal, campaignType is `CONVERSIONS`
</ParamField>

<ParamField body="carousel" type="array">
	<Expandable title="Carousel Array">
		<ParamField body="name" type="string" required>
			Carousel Name
		</ParamField>
		<ParamField body="description" type="string">
			Carousel Description
		</ParamField>
		<ParamField body="picture" type="string">
			Carousel Picture URL
		</ParamField>
		<ParamField body="video_id" type="string">
			Carousel Video ID
		</ParamField>
		<ParamField body="link" type="string">
			Carousel Link URL
		</ParamField>
	</Expandable>

</ParamField>

<ParamField body="primaryText" type="array">
	<Note>Use this with carousel ads only</Note>
	<Expandable title="Primary Text Array">
		<ParamField body="text" type="string" required>
			Main ad copy text
		</ParamField>
	</Expandable>
</ParamField>

<ParamField body="isDynamicCreative" type="boolean" >
	Whether it's dynamic creative, useable with images/videos

    <Note>
    	Use `isDynamicCreative` = `false`, to create multiple standard
    	`non-dynamic` ads in a single AdSet.
    </Note>

</ParamField>

<ParamField body="multipleCreatives" type="array">

    <Expandable title="Multiple Creative Configuration">
    	<ParamField body="url" type="string" required>
    		Destination URL
    	</ParamField>

    	<ParamField body="callToAction" type="string" required>
    		Supported call to action types are `LEARN_MORE`, `SHOP_NOW`, `SIGN_UP`, `DOWNLOAD`,
    		`WATCH_MORE`, `APPLY_NOW`, `SUBSCRIBE`, `GET_QUOTE`, `CONTACT_US`, `ORDER_NOW`,
    		`REQUEST_TIME`, `GET_SHOWTIMES`, `LISTEN_NOW`, `SEE_MENU`, `BOOK_TRAVEL`
    	</ParamField>

    	<ParamField body="primaryText" type="string" required>
    		Primary Text
    	</ParamField>

    	<ParamField body="headline1" type="string">
    		Headline
    	</ParamField>

    	<ParamField body="image" type="string">
    		Image URL
    	</ParamField>

    	<ParamField body="video" type="object">
    		<Expandable title="Video Data">
    			<ParamField body="video_id" type="string">
    				Video ID
    			</ParamField>
    			<ParamField body="thumbnail_url" type="string">
    				Thumbnail URL
    			</ParamField>
    		</Expandable>
    	</ParamField>

    	<ParamField body="phoneNumber" type="string">
    		Phone Number (for Call Ads)
    	</ParamField>

    	<ParamField
    		body="greetingText"
    		type="string"
    	>
    		Greeting Text
    	</ParamField>

    	<ParamField body="conversations" type="array">
    		<Expandable title="Conversations Array">
    			<ParamField body="title" type="string" required>
    				Conversation Title
    			</ParamField>
    		</Expandable>
    	</ParamField>

    </Expandable>

</ParamField>

<ParamField body="pixelId" type="string" required>
	Facebook Pixel ID for conversion tracking
</ParamField>

<ParamField body="conversionGoal" type="string" required>
	Campaign conversion goal type, get the list of conversion goals from
	[here](/api-reference/endpoint/meta/get_custom_conversions), use
	`CONVERSIONS` in campaignType to get the list of conversion goals.
</ParamField>

<ParamField body="callToAction" type="string">
	<Note>Use this with carousel ads only</Note>
	Supported call to action types are `LEARN_MORE`, `SHOP_NOW`, `SIGN_UP`, `DOWNLOAD`,
	`WATCH_MORE`, `APPLY_NOW`, `SUBSCRIBE`, `GET_QUOTE`, `CONTACT_US`, `ORDER_NOW`,
	`REQUEST_TIME`, `GET_SHOWTIMES`, `LISTEN_NOW`, `SEE_MENU`, `BOOK_TRAVEL`
</ParamField>
