### Boost Post Ad Parameters

<ParamField body="campaignType" type="string" required>
	For Boost Post Goal, campaignType is `POST_ENGAGEMENT`
</ParamField>

<ParamField body="conversations" type="array">
	<Expandable title="Conversations Array">
		<ParamField body="title" type="string" required>
			Conversation Title
		</ParamField>
	</Expandable>

    <Note>
    	Only For Messenger Use and When Call to Action is `Send Message`
    </Note>

</ParamField>

<ParamField body="greetingText" type="string">
	Greeting Text

    <Note>
    	Use When Call to Action is `Send Message`
    </Note>

</ParamField>

<ParamField body="messagingApps" type="array" >
	Messaging apps = `MESSENGER`, `INSTAGRAM`, `WHATSAPP`

    <Note>
    	Use When Call to Action is `Send Message`
    </Note>

</ParamField>

<ParamField body="pagePostId" type="string" required>
	Facebook page post ID, you can get post id from
	[here](/api-reference/endpoint/meta/get_page_posts)
</ParamField>

<ParamField body="callToAction" type="string" required>
	Call to Action, Supported Call to Action is `Send Message`
</ParamField>
